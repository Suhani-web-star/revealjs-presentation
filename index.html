Sample: Net Profit in Python

from dataclasses import dataclass

@dataclass
class PrintRun:
    copies: int
    price: float
    cost_per_copy: float

    def profit(self, returns_rate=0.08):
        sold = int(self.copies * (1 - returns_rate))
        revenue = sold * self.price
        cost = self.copies * self.cost_per_copy
        return revenue - cost

run = PrintRun(copies=120000, price=6.0, cost_per_copy=3.9)
print(f"Estimated profit: ₹{run.profit():,.0f}")

\[\mathbb{E}[R_i] = R_f + \beta_i\big(\mathbb{E}[R_m] - R_f\big)\]
